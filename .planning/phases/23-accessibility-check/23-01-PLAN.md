```typescript
export interface LinkWarning {
  href: string
  text: string
  lineNumber: number
  problem: 'empty-href' | 'hash-placeholder' | 'example-domain' | 'missing-protocol'
  severity: 'error' | 'warning'
  message: string
}

function lineOf(html: string, needle: string): number {
  const idx = html.indexOf(needle)
  if (idx === -1) return 0
  return html.slice(0, idx).split('\n').length
}

export function analyzeLinkProblems(html: string): LinkWarning[]
```

<feature>
  <name>analyzeAccessibility pure function</name>
  <files>lib/engine/analyzeAccessibility.ts, lib/engine/analyzeAccessibility.test.ts</files>
  <behavior>
    missing-alt detection (per D-01.1):
    - `<img src="x.jpg">` (no alt attribute) => 1 warning, type='missing-alt', severity='error'
    - `<img src="x.jpg" alt="">` (decorative) => 0 warnings
    - `<img src="x.jpg" alt="photo">` => 0 warnings
    - Multiple img without alt => warning per img, each with correct lineNumber

    low-contrast detection (per D-01.2):
    - Element with `style="color:#000;background-color:#fff"` (ratio 21:1) => 0 warnings
    - Element with `style="color:#777;background-color:#888"` (low contrast) => 1 warning, type='low-contrast'
    - Element with only `color` but no `background-color` in same style => 0 warnings (skip)
    - Unparseable color format (named colors, hsl) => 0 warnings (skip silently)
    - Warning detail includes fg, bg, ratio, required

    heading-skip detection (per D-01.3):
    - `<h1>A</h1><h3>B</h3>` (h1->h3 skip) => 1 warning, type='heading-skip', severity='warning'
    - `<h1>A</h1><h2>B</h2>` (sequential) => 0 warnings
    - `<h2>A</h2><h4>B</h4>` (h2->h4 skip) => 1 warning
    - `<h3>A</h3><h2>B</h2>` (going back up) => 0 warnings (not a skip)
    - First heading being h2 (no h1) => 0 warnings (per Research pitfall 5)

    summary (per D-05):
    - Returns AccessibilitySummary { warnings, passCount, failCount }
    - passCount = total checked items - failCount
    - failCount = warnings.length

    WCAG formula correctness:
    - relativeLuminance(255,255,255) => 1.0
    - relativeLuminance(0,0,0) => 0.0
    - contrastRatio(white, black) => 21:1
    - Uses 0.04045 threshold (WCAG 2.1, NOT 0.03928)
  </behavior>
  <implementation>
    After all tests pass, the implementation should:
    1. Export AccessibilityWarning interface with type/severity/message/lineNumber/detail fields
    2. Export AccessibilitySummary interface with warnings/passCount/failCount
    3. Implement WCAG relative luminance: linearize(c) using 0.04045 threshold, relativeLuminance(r,g,b) with 0.2126/0.7152/0.0722 coefficients
    4. Implement contrastRatio(l1,l2) = (lighter+0.05)/(darker+0.05)
    5. Implement parseColor supporting #RGB, #RRGGBB, rgb(r,g,b) — return null for unsupported formats
    6. Implement extractCssProperty(style, prop) to extract a single CSS property value from inline style string
    7. Main analyzeAccessibility(html) function using cheerio:
       - $('img').each() for alt detection (undefined check, not falsy check)
       - $('h1,h2,h3,h4,h5,h6') collection and sequential diff>1 detection
       - $('[style]').each() for color/background-color pair extraction and contrast check
       - Use lineOf() helper (same as analyzeLinkProblems.ts)
    8. Apply 4.5:1 threshold for all text (simplified per Research open question 1)
    9. Calculate passCount = (img count) + (heading pairs checked) + (color pairs found) - warnings.length
  </implementation>
