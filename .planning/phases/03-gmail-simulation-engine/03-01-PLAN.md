    ```typescript
    // Conditional <style> block processing (per D-03)
    if (ruleset.styleBlockBehavior) {
      // Gmail path: inspect each <style> block, strip only if disallowed pattern found
      const patterns = ruleset.styleBlockBehavior.disallowedPatterns.map(
        p => new RegExp(p, 'i')
      )
      const toRemove: ReturnType<typeof $.find>[] = []
      $('style').each((_, el) => {
        const cssText = $(el).text()
        if (patterns.some(re => re.test(cssText))) {
          toRemove.push($(el))
        }
      })
      toRemove.forEach(el => el.remove())
    } else if (ruleset.stripHeadStyles) {
    ```
